# Basic test structure (TypeScript)
import { test, expect } from '@playwright/test';

test('Login test', async ({ page }) => {
  await page.goto('https://example.com');
  await page.fill('#username', 'admin');
  await page.fill('#password', 'secret');
  await page.click('button[type=submit]');
  await expect(page).toHaveURL(/dashboard/);
});

CSS & XPath

// CSS selectors
page.locator('#id')
page.locator('.class')
page.locator('button[type=submit]')
page.locator('div.card > h2')
page.locator('input[name=email]')

// XPath
page.locator('xpath=//button[@id="submit"]')
page.locator('xpath=//div[contains(@class,"modal")]')

// Filter & chain locators
page.getByRole('listitem').filter({ hasText: 'Product' })
page.locator('.row').nth(2)
page.locator('.item').first()
page.locator('.item').last()

⏱️ Waits

// Auto-waiting — Playwright waits automatically for most actions
// But when you need explicit waits:

// Wait for element
await page.waitForSelector('.modal');
await page.waitForSelector('.modal', { state: 'visible' });
await page.waitForSelector('.loader', { state: 'hidden' });

// Wait for navigation
await page.waitForURL('**/dashboard');
await page.waitForLoadState('networkidle');
await page.waitForLoadState('domcontentloaded');

// Wait for response
const [response] = await Promise.all([
  page.waitForResponse('**/api/users'),
  page.click('#loadBtn')
]);

// Wait for function
await page.waitForFunction(() => document.querySelectorAll('.item').length > 5);

// Custom timeout
await page.click('#btn', { timeout: 10000 });

🌍 API Testing

import { test, expect } from '@playwright/test';

// GET request
test('GET users', async ({ request }) => {
  const response = await request.get('https://api.example.com/users');
  expect(response.status()).toBe(200);
  const body = await response.json();
  expect(body.length).toBeGreaterThan(0);
});

// POST request
test('Create user', async ({ request }) => {
  const response = await request.post('https://api.example.com/users', {
    data: { name: 'John', email: 'john@test.com' },
    headers: { 'Authorization': 'Bearer token123' }
  });
  expect(response.status()).toBe(201);
});

// API context with base URL
const apiContext = await playwright.request.newContext({
  baseURL: 'https://api.example.com',
  extraHTTPHeaders: { 'Authorization': `Bearer ${token}` }
});

🌐 Network Interception

// Mock API response
await page.route('**/api/users', route => route.fulfill({
  status: 200,
  body: JSON.stringify([{ id: 1, name: 'Mock User' }])
}));

// Abort requests
await page.route('**/*.png', route => route.abort());

// Modify request
await page.route('**/api/**', route => route.continue({
  headers: { ...(route.request().headers()), 'X-Custom-Header': 'test' }
}));

// Listen to responses
page.on('response', response => {
  if (response.url().includes('/api')) console.log(response.status());
});

💡 Pro Tips

🎯 Use getByRole First

Always prefer getByRole, getByLabel, getByTestId over CSS/XPath. They are more resilient to UI changes.

🔁 Use Fixtures

Create custom fixtures for logged-in state, API context, and common setups. Avoids repeating login in every test.

📹 Record Traces

Enable trace:'on-first-retry' in config. Traces capture screenshots, DOM snapshots and network logs for debugging.

🚀 Page Object Model

Create a class for each page with locators and methods. Makes tests readable and easy to maintain when UI changes.

⚡ Parallel by Default

Playwright runs tests in parallel. Make tests independent — no shared state. Use test.describe.serial() only when needed.

🎭 Use codegen

Run npx playwright codegen to record tests. Great for getting initial locators. Always clean up generated code after.
